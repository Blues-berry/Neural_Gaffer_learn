1. 临时禁用全局 Python 库路径
这行命令会让 Python 假装看不见 AppData\Roaming 下那个坏掉的 TensorFlow。

PowerShell

$env:PYTHONNOUSERSITE = "1"
2. 文件夹补齐

$base = "./real_data_relighting"
$fix = "preprocessed_lighting_data"
$models = @("Mandalorian_helmet", "dragon", "duck", "statue_of_hand", "vege_dog")
 
 
foreach ($m in $models) {
    # 补齐 input_image 下的幽灵文件夹
    $d1 = "$base/$m/input_image/$fix"
    if (!(Test-Path $d1)) { New-Item -ItemType Directory -Path $d1 -Force }

    # 补齐 pred_image 下的幽灵文件夹
    $d2 = "$base/$m/pred_image/$fix"
    if (!(Test-Path $d2)) { New-Item -ItemType Directory -Path $d2 -Force }

    # 补齐 target_envmap_ldr 下的幽灵文件夹
    $d3 = "$base/$m/target_envmap_ldr/$fix"
    if (!(Test-Path $d3)) { New-Item -ItemType Directory -Path $d3 -Force }
}
Write-Host "所有幽灵文件夹已补齐！" -ForegroundColor Green
